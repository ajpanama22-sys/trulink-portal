  <script>
    const form = document.getElementById("registerForm");
    const emailInput = document.getElementById("email");
    const errorMsg = document.getElementById("error");

    form.addEventListener("submit", function(e) {
      e.preventDefault();
      const email = emailInput.value;
      const dominio = email.split("@")[1];
      const bloqueados = ["gmail.com","yahoo.com","hotmail.com","outlook.com"];
      if (bloqueados.includes(dominio.toLowerCase())) {
        errorMsg.textContent = "Debe usar un correo corporativo con dominio propio.";
      } else {
        errorMsg.textContent = "";
        alert("Registro enviado para validación interna.");
      }
    });
  </script>
